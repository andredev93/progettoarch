	int codice_prodotto;	4 byte

}

ricerca(id) {
	cerca prodotto o posizione in cui inserire un prodotto
	v0 = 0/1
	v1 = indirizzo prodotto
}

inserisci(id, nome) {
	ricerca(id)
	sposta_memoria_avanti(id + 1)
	inserisci
}

cancella(id) {
	ricerca(id)
	cancella
	sposta_memoria_indietro(id + 1)
}

aumenta(id) {
	ricerca(id)
	controlla_tetto
}

diminuisci(id) {
	ricerca(id)
	controllo vuoto
}
